 20    Returns:
 21        string: LaTeX string
 22    """
 23
 24    if (n==0):
 25        return '0'
 26    if (np.abs(n)>10**(1-prec)):
 27        n_latex = f'${{:.{prec}g}}$'.format(num2tex(n,precision=prec))
